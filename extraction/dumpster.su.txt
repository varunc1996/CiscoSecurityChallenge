    function get_results() {
        var query = $("#query").val();
        var dataString = 'query=' + query;
        $.ajax({
            type: "POST",
            url: "http://hyipzmonitor.com/common/scripts/search_handler.php",
            data: dataString,
            cache: false,
            success: function (result) {
                $("#results").html(result);
            }
        });
    }

    function mark(caller) {
        var cookie = $(caller).attr("data-request");
        var dataString = 'cookie=' + cookie + ' && id=' + caller.id;
//        alert(dataString);
        $.ajax({
            beforeSend: function (xhr) {
                $(caller).text("Requesting.....");
            },
            type: "POST",
            url: "mark.php",
            data: dataString,
            cache: false,
            success: function (response) {
                $(caller).html(response);
                $(caller).removeAttr("onclick");
                $(caller).attr("disabled", "");
            }
        });
    }
